54//! QKlipperConsole class
55/*!
56 This class is responsible for the actual communication to the klipper instance via moonraker.
57 It provides methods for the moonraker API for controlling and querying aspects of the klipper
58 instance.
59
60 Prior to connecting, it is required to pass instance references to QKlipperPrinter, QKlipperSystem
61 and QKlipperServer that have been setup and configured.
62*/

68 //! ConnectionState enum
69 /*!
70 This enum provides flags for the various connection states of the console
71 */
72 enum ConnectionState
73 {
74 Idle = 0, /*!!< Connection state is idle. This is the default state */
75 Startup = 0b00000001, /*!!< Connection has been established and currently syncing */
76 Connecting = 0b00000010, /*!!< Connecting to websocket address */
77 Connected = 0b00000100, /*!!< Connection to websocket established */
78 MoonrakerConnected = 0b00001000, /*!!< Connection to moonraker established */
79 KlipperConnected = 0b00010000, /*!!< Connection to klipper verified */
80 Syncronized = 0b00100000, /*!!< Connection to websocket, moonraker and klipper established and startup completed */
81 Error = 0b10000000 /*!!< Error detected with the connection */
82 };
