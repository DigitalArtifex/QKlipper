19#ifndef QKLIPPERENDSTOPSTATUS_H
20#define QKLIPPERENDSTOPSTATUS_H
21
22#include <QObject>
23#include <QMap>
24
25//! QKlipperEndstopStatus class
26/*!
27 This class provides methods to read the status of all
28 end stops sent by klipper in printer.query_endstops
29*/
30class QKlipperEndstopStatus
31{
32 Q_GADGET
33public:
34 /*!
35 * Subscript operator
36 *
37 * \param name The name of the endstop (eg "x")
38 *
39 * \returns QKlipperEndstopStatus reference
40 */
41 bool &operator[](const QString &name)
42 {
43 return m_endstops[name];
44 }
45
46 bool operator==(const QKlipperEndstopStatus &value)
47 {
48 QStringList keys = value.m_endstops.keys();
49
50 foreach(QString key, keys)
51 {
52 if(!m_endstops.contains(key))
53 return false;
54 else if(m_endstops[key] != value.endStop(key))
55 return false;
56 }
57
58 return true;
59 }
60
61 bool operator!=(const QKlipperEndstopStatus &value)
62 {
63 return !(this == &value);
64 }
65
66 /*!
67 * Convenience function to return the 'x' endstop status
68 *
69 * \returns QKlipperEndstopStatus reference
70 */
71 bool x() const
72 {
73 return m_endstops["x"];
74 }
75
76 /*!
77 * Convenience function to return the 'y' endstop status
78 *
79 * \returns QKlipperEndstopStatus reference
80 */
81 bool y() const
82 {
83 return m_endstops["y"];
84 }
85
86 /*!
87 * Convenience function to return the 'z' endstop status
88 *
89 * \returns QKlipperEndstopStatus reference
90 */
91 bool z() const
92 {
93 return m_endstops["z"];
94 }
95
96 /*!
97 * Convenience function to return the endstop status by name
98 *
99 * \param name The name of the endstop
100 *
101 * \returns QKlipperEndstopStatus reference
102 */
103 bool endStop(const QString &name) const
104 {
105 return m_endstops[name];
106 }
107
108 /*!
109 * Checks to see if there is a stored reference for the endstop name
110 *
111 * \param name The name of the endstop
112 *
113 * \returns True if the reference exists
114 */
115 bool contains(const QString &name) const
116 {
117 return m_endstops.contains(name);
118 }
119
120private:
121 QMap<QString, bool> m_endstops;
122};
123
124Q_DECLARE_METATYPE(QKlipperEndstopStatus)
125#endif // QKLIPPERENDSTOPSTATUS_H
